9static OPERATOR_LINE_REGEX: Lazy<Regex> = Lazy::new(|| {
10    Regex::new(r"([A-Z_]+)\s*\(plan_node_id=(-?\d+)(?:\s*\(operator\s+id=(\d+)\))?\)").unwrap()

15});

20    pub fn parse_advanced(input: &str) -> Result<Profile, String> {
21        let mut profile = StarRocksProfileParser::parse(input)?;
22
23        // 构建高级的执行树
24        let execution_tree = Self::build_execution_tree(input)?;
25        profile.execution_tree = Some(execution_tree);
26
27        Ok(profile)
28    }

30    pub fn build_execution_tree(profile_text: &str) -> Result<ExecutionTree, String> {
31        let mut nodes = Vec::new();
32        let mut node_map = HashMap::new();
33
34        // 分割Profile文本为操作符块
35        let operator_blocks = Self::split_into_operator_blocks(profile_text)?;
36
37        // 解析每个操作符块
38        for block in operator_blocks {
39            let node = Self::parse_operator_block(&block)?;
40            node_map.insert(node.plan_node_id.unwrap_or(-1), node.clone());
41            nodes.push(node);
42        }
43
44        // 构建树结构
45        Self::build_tree_relationships(&mut nodes, &node_map);
46
47        // 确定根节点（通常是第一个节点）
48        if nodes.is_empty() {
49            return Err("No operators found in profile".to_string());
50        }
51
52        let root = nodes[0].clone();
53
54        Ok(ExecutionTree { root, nodes })
55    }

57    fn split_into_operator_blocks(profile_text: &str) -> Result<Vec<String>, String> {
58        let lines: Vec<&str> = profile_text.lines().collect();
59        let mut blocks = Vec::new();
60        let mut current_block = String::new();
61        let mut in_operator_block = false;
62
63        for line in lines {
64            let trimmed = line.trim();
65
66            // 检查是否是操作符开始行
67            if Self::is_operator_start_line(trimmed) {
68                // 保存之前的块
69                if !current_block.is_empty() {
70                    blocks.push(current_block);
71                }
72                current_block = line.to_string();
73                in_operator_block = true;
74            } else if in_operator_block {
75                current_block.push('\n');
76                current_block.push_str(line);
77
78                // 检查是否是操作符结束（下一行是空行或者开始新的操作符）
79                if trimmed.is_empty() && !current_block.trim().is_empty() {
80                    in_operator_block = false;
81                }
82            }
83        }
84
85        if !current_block.is_empty() {
86            blocks.push(current_block);
87        }
88
89        Ok(blocks)
90    }

92    fn is_operator_start_line(line: &str) -> bool {
93        // 根据StarRocks Profile格式，操作符行格式如：
94        // "RESULT_SINK (plan_node_id=-1):"
95        // "CHUNK_ACCUMULATE (plan_node_id=-1):"
96        // "LIMIT (plan_node_id=1) (operator id=1):"
97        OPERATOR_LINE_REGEX.is_match(line)
98    }

100    fn parse_operator_block(block: &str) -> Result<ExecutionTreeNode, String> {
101        let lines: Vec<&str> = block.lines().collect();
102
103        if lines.is_empty() {
104            return Err("Empty operator block".to_string());
105        }
106
107        // 解析操作符标题行
108        let first_line = lines[0].trim();
109        let (operator_name, plan_node_id, operator_id) = Self::parse_operator_header(first_line)?;
110
111        // 解析指标
112        let metrics = Self::parse_operator_metrics(&lines[1..])?;
113
114        // 确定节点类型
115        let node_type = Self::determine_node_type(&operator_name);
116
117        let node = ExecutionTreeNode {
118            id: format!("node_{}", plan_node_id.unwrap_or(-1)),
119            operator_name: operator_name.clone(),
120            node_type,
121            plan_node_id,
122            parent_plan_node_id: None, // 稍后构建树关系时设置
123            metrics,
124            children: Vec::new(),
125            depth: 0,
126            is_hotspot: false, // 默认值，将在分析时设置
127            hotspot_severity: HotSeverity::Normal,
128        };
129
130        Ok(node)
131    }

133    fn parse_operator_header(line: &str) -> Result<(String, Option<i32>, Option<String>), String> {
134        let captures = OPERATOR_LINE_REGEX.captures(line)
135            .ok_or_else(|| format!("Invalid operator header: {}", line))?;
136
137        let operator_name = captures.get(1)
138            .ok_or("Missing operator name")?
139            .as_str()
140            .to_string();
141
142        let plan_node_id = captures.get(2)
143            .and_then(|m| m.as_str().parse::<i32>().ok());
144
145        let operator_id = captures.get(3)
146            .map(|m| m.as_str().to_string());
147
148        Ok((operator_name, plan_node_id, operator_id))
149    }

151    fn parse_operator_metrics(lines: &[&str]) -> Result<OperatorMetrics, String> {
152        let mut metrics = OperatorMetrics {
153            operator_total_time: None,
154            push_chunk_num: None,
155            push_row_num: None,
156            pull_chunk_num: None,
157            pull_row_num: None,
158            push_total_time: None,
159            pull_total_time: None,
160            memory_usage: None,
161            output_chunk_bytes: None,
162            specialized: OperatorSpecializedMetrics::None,
163        };
164
165        for line in lines {
166            if let Some((key, value)) = Self::parse_metric_line(line) {
167                Self::set_metric_value(&mut metrics, &key, &value);
168            }
169        }
170
171        Ok(metrics)
172    }

174    fn parse_metric_line(line: &str) -> Option<(String, String)> {
175        METRIC_LINE_REGEX.captures(line)
176            .and_then(|caps| {
177                let key = caps.get(1)?.as_str().replace(" ", "");
178                let value = caps.get(2)?.as_str().to_string();
179                Some((key, value))
180            })
181    }

183    fn set_metric_value(metrics: &mut OperatorMetrics, key: &str, value: &str) {
184        match key {
185            "OperatorTotalTime" => {
186                metrics.operator_total_time = Self::parse_duration(value).ok();
187            }
188            "PushChunkNum" => {
189                metrics.push_chunk_num = Self::parse_number(value).ok();
190            }
191            "PushRowNum" => {
192                metrics.push_row_num = Self::parse_number(value).ok();
193            }
194            "PullChunkNum" => {
195                metrics.pull_chunk_num = Self::parse_number(value).ok();
196            }
197            "PullRowNum" => {
198                metrics.pull_row_num = Self::parse_number(value).ok();
199            }
200            "PushTotalTime" => {
201                metrics.push_total_time = Self::parse_duration(value).ok();
202            }
203            "PullTotalTime" => {
204                metrics.pull_total_time = Self::parse_duration(value).ok();
205            }
206            "OutputChunkBytes" => {
207                metrics.output_chunk_bytes = Self::parse_bytes(value).ok();
208            }
209            // 处理专用指标，这里根据操作符类型处理
210            _ => {}
211        }
212    }

214    fn parse_duration(value: &str) -> Result<Duration, String> {
215        // 解析时间格式，如 "1h30m", "5s499ms", "0ns"
216        if value.contains("h") {
217            let hours: f64 = value.split("h").next()
218                .ok_or("Invalid hours format")?
219                .parse()
220                .map_err(|_| "Invalid hours number")?;
221            let minutes: f64 = value.split("h").nth(1)
222                .unwrap_or("0m")
223                .trim_end_matches("m")
224                .parse()
225                .unwrap_or(0.0);
226            Ok(Duration::from_millis(((hours * 3600.0 + minutes * 60.0) * 1000.0) as u64))
227        } else if value.contains("m") {
228            let minutes: f64 = value.split("m").next()
229                .ok_or("Invalid minutes format")?
230                .parse()
231                .map_err(|_| "Invalid minutes number")?;
232            Ok(Duration::from_millis((minutes * 60.0 * 1000.0) as u64))
233        } else if value.contains("s") {
234            let seconds: f64 = value.split("s").next()
235                .ok_or("Invalid seconds format")?
236                .parse()
237                .map_err(|_| "Invalid seconds number")?;
238            Ok(Duration::from_millis((seconds * 1000.0) as u64))
239        } else if value.contains("ms") {
240            let ms: f64 = value.split("ms").next()
241                .ok_or("Invalid ms format")?
242                .parse()
243                .map_err(|_| "Invalid ms number")?;
244            Ok(Duration::from_millis(ms as u64))
245        } else if value.contains("us") || value.contains("μs") {
246            let us: f64 = value.split("us").next()
247                .unwrap_or(value.trim_end_matches("μs"))
248                .parse()
249                .map_err(|_| "Invalid us number")?;
250            Ok(Duration::from_micros(us as u64))
251        } else if value.contains("ns") {
252            let ns: f64 = value.split("ns").next()
253                .ok_or("Invalid ns format")?
254                .parse()
255                .map_err(|_| "Invalid ns number")?;
256            Ok(Duration::from_nanos(ns as u64))
257        } else {
258            Err(format!("Unknown duration format: {}", value))
259        }
260    }

262    fn parse_number(value: &str) -> Result<u64, String> {
263        value.replace(",", "").parse().map_err(|_| format!("Invalid number: {}", value))
264    }

266    fn parse_bytes(value: &str) -> Result<u64, String> {
267        // 解析字节格式，如 "2.174K (2174)", "1.463 KB", "18.604 MB"
268        let clean_value = value.split_whitespace().next().unwrap_or(value);
269
270        if clean_value.contains("GB") {
271            let gb: f64 = clean_value.replace("GB", "").parse().map_err(|_| "Invalid GB format")?;
272            Ok((gb * 1024.0 * 1024.0 * 1024.0) as u64)
273        } else if clean_value.contains("MB") {
274            let mb: f64 = clean_value.replace("MB", "").parse().map_err(|_| "Invalid MB format")?;
275            Ok((mb * 1024.0 * 1024.0) as u64)
276        } else if clean_value.contains("KB") || clean_value.contains("K") {
277            let kb: f64 = clean_value.replace("KB", "").replace("K", "").parse().map_err(|_| "Invalid KB format")?;
278            Ok((kb * 1024.0) as u64)
279        } else {
280            clean_value.replace(",", "").parse().map_err(|_| format!("Invalid bytes format: {}", value))
281        }
282    }

284    fn determine_node_type(operator_name: &str) -> NodeType {
285        match operator_name {
286            "OLAP_SCAN" => NodeType::OlapScan,
287            "CONNECTOR_SCAN" => NodeType::ConnectorScan,
288            "HASH_JOIN" | "NL_JOIN" | "CROSS_JOIN" => NodeType::HashJoin,
289            "AGGREGATE" => NodeType::Aggregate,
290            "LIMIT" => NodeType::Limit,
291            "EXCHANGE_SINK" => NodeType::ExchangeSink,
292            "EXCHANGE_SOURCE" => NodeType::ExchangeSource,
293            "RESULT_SINK" => NodeType::ResultSink,
294            "CHUNK_ACCUMULATE" => NodeType::ChunkAccumulate,
295            "SORT" => NodeType::Sort,
296            _ => NodeType::Unknown,
297        }
298    }
